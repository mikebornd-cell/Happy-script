function SetInfAmmo() local c=player.Character if not c then return end for _,o in pairs(c:GetDescendants()) do if o:IsA("IntValue") or o:IsA("NumberValue") then local n=string.lower(o.Name or "") if n:find("ammo") or n:find("clip") or n:find("mag") or n:find("bullet") then o.Value=9999 end end end for _,t in pairs(player.Backpack:GetChildren()) do for _,o in pairs(t:GetDescendants()) do if o:IsA("IntValue") or o:IsA("NumberValue") then local n=string.lower(o.Name or "") if n:find("ammo") or n:find("clip") or n:find("mag") or n:find("bullet") then o.Value=9999 end end end end end
function EnableInfAmmo() if infAmmo then return end infAmmo=true SetInfAmmo() infLoop=game:GetService("RunService").Heartbeat:Connect(function() if infAmmo then SetInfAmmo() end end) Rayfield:Notify({Title="Infinite Ammo",Content="ON",Duration=1}) end
function DisableInfAmmo() infAmmo=false if infLoop then infLoop:Disconnect() end Rayfield:Notify({Title="Infinite Ammo",Content="OFF",Duration=1}) end
function ApplyRapid(t) for _,v in pairs(t:GetDescendants()) do if v:IsA("NumberValue") or v:IsA("IntValue") then local n=string.lower(v.Name or "") if n:find("firerate") or n:find("delay") then v.Value=0 end if n:find("rpm") then v.Value=999999 end end end end
function EnableRapid() if rapidFire then return end rapidFire=true for _,t in pairs(player.Backpack:GetChildren()) do ApplyRapid(t) end for _,t in pairs(player.Character:GetChildren()) do ApplyRapid(t) end rapidConn[1]=player.Backpack.ChildAdded:Connect(ApplyRapid) rapidConn[2]=player.Character.ChildAdded:Connect(ApplyRapid) Rayfield:Notify({Title="Rapid Fire",Content="ON",Duration=1}) end
function DisableRapid() rapidFire=false for _,c in pairs(rapidConn) do if c then pcall(function() c:Disconnect() end) end end rapidConn={} Rayfield:Notify({Title="Rapid Fire",Content="OFF",Duration=1}) end
function ApplyNoRecoil(t) for _,v in pairs(t:GetDescendants()) do if v:IsA("NumberValue") or v:IsA("IntValue") then local n=string.lower(v.Name or "") if n:find("recoil") or n:find("kick") then v.Value=0 local conn=v:GetPropertyChangedSignal("Value"):Connect(function() if v.Value~=0 then v.Value=0 end end) table.insert(recoilConn,conn) end end end end
function EnableNoRecoil() if noRecoil then return end noRecoil=true for _,t in pairs(player.Backpack:GetChildren()) do ApplyNoRecoil(t) end for _,t in pairs(player.Character:GetChildren()) do ApplyNoRecoil(t) end recoilConn[1]=player.Backpack.ChildAdded:Connect(ApplyNoRecoil) recoilConn[2]=player.Character.ChildAdded:Connect(ApplyNoRecoil) Rayfield:Notify({Title="No Recoil",Content="ON",Duration=1}) end
function DisableNoRecoil() noRecoil=false for _,c in pairs(recoilConn) do if c then pcall(function() c:Disconnect() end) end end recoilConn={} Rayfield:Notify({Title="No Recoil",Content="OFF",Duration=1}) end
function SetSpeed() local c=player.Character if c and c:FindFirstChild("Humanoid") then c.Humanoid.WalkSpeed=speedVal end Rayfield:Notify({Title="Speed",Content="Set to "..speedVal, Duration=1}) end
function EnableSpeed() if speedOn then return end speedOn=true speedLoop=game:GetService("RunService").Heartbeat:Connect(function() if speedOn then local c=player.Character if c and c:FindFirstChild("Humanoid") then c.Humanoid.WalkSpeed=speedVal end end end) Rayfield:Notify({Title="Speed Modifier",Content="ON",Duration=1}) end
function DisableSpeed() speedOn=false if speedLoop then speedLoop:Disconnect() speedLoop=nil end local c=player.Character if c and c:FindFirstChild("Humanoid") then c.Humanoid.WalkSpeed=16 end Rayfield:Notify({Title="Speed Modifier",Content="OFF",Duration=1}) end

function FindTarget(n) if not n then return nil end for _,o in pairs(workspace:GetDescendants()) do if o:IsA("Model") and o.Name==n then if o:FindFirstChild("HumanoidRootPart") or o:FindFirstChild("Head") then return o end end end return nil end
function LockCameraTo(t) if cameraLocked then if cameraLoop then cameraLoop:Disconnect() end end cameraLocked=true cameraLoop=game:GetService("RunService").RenderStepped:Connect(function() if not cameraLocked then return end local target=FindTarget(t) if target then local part=target:FindFirstChild("HumanoidRootPart") or target:FindFirstChild("Head") if part then camera.CFrame=camera.CFrame:Lerp(CFrame.new(camera.CFrame.Position,part.Position),smoothness) end end end) Rayfield:Notify({Title="Camera Lock",Content="Locked to "..t,Duration=2}) end
function UnlockCam() cameraLocked=false if cameraLoop then cameraLoop:Disconnect() cameraLoop=nil end Rayfield:Notify({Title="Camera Lock",Content="Unlocked",Duration=2}) end
function LockNearest() local c=player.Character if not c then return end local r=c:FindFirstChild("HumanoidRootPart") if not r then return end local nearest,shortest=nil,math.huge for _,o in pairs(workspace:GetDescendants()) do if o:IsA("Model") and o~=c then local p=o:FindFirstChild("HumanoidRootPart") or o:FindFirstChild("Head") if p then local d=(r.Position-p.Position).Magnitude if d<shortest and d<150 then shortest=d nearest=o end end end end if nearest then LockCameraTo(nearest.Name) else Rayfield:Notify({Title="Camera Lock",Content="No enemies nearby",Duration=2}) end end
