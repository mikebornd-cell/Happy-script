local function createTab(name, order)
    local page = createTabPage(name) local btn = Instance.new("TextButton") btn.Name = name btn.Size = UDim2.new(1, 0, 0, 28) btn.BackgroundColor3 = C.ElementBg btn.BackgroundTransparency = 0.5 btn.Text = "" btn.LayoutOrder = order btn.ZIndex = 3 btn.Parent = TabContainer corner(btn, 4)
    local label = Instance.new("TextLabel") label.Size = UDim2.new(1, -10, 1, 0) label.Position = UDim2.new(0, 10, 0, 0) label.BackgroundTransparency = 1 label.Text = name label.TextColor3 = C.TextSecondary label.TextSize = 12 label.Font = FONT.Label label.TextXAlignment = Enum.TextXAlignment.Left label.ZIndex = 4 label.Parent = btn
    local indicator = Instance.new("Frame") indicator.Name = "Indicator" indicator.Size = UDim2.new(0, 2, 0.6, 0) indicator.Position = UDim2.new(0, 0, 0.2, 0) indicator.BackgroundColor3 = C.Accent indicator.BorderSizePixel = 0 indicator.Visible = false indicator.ZIndex = 5 indicator.Parent = btn corner(indicator, 1)
    tabs[name] = {page = page, btn = btn, label = label, indicator = indicator}
    btn.MouseEnter:Connect(function() if activeTab ~= name then tween(btn, {BackgroundTransparency = 0.3}, 0.1):Play() tween(label, {TextColor3 = C.TextPrimary}, 0.1):Play() end end)
    btn.MouseLeave:Connect(function() if activeTab ~= name then tween(btn, {BackgroundTransparency = 0.5}, 0.1):Play() tween(label, {TextColor3 = C.TextSecondary}, 0.1):Play() end end)
    btn.MouseButton1Click:Connect(function() if activeTab and tabs[activeTab] then tabs[activeTab].page.Visible = false tabs[activeTab].indicator.Visible = false tween(tabs[activeTab].btn, {BackgroundTransparency = 0.5}, 0.1):Play() tween(tabs[activeTab].label, {TextColor3 = C.TextSecondary}, 0.1):Play() end activeTab = name page.Visible = true indicator.Visible = true tween(btn, {BackgroundTransparency = 0.2}, 0.15):Play() tween(label, {TextColor3 = C.Accent}, 0.15):Play() end)
    return page
end

    local state = false local btn = Instance.new("TextButton") btn.Size = UDim2.new(1, 0, 1, 0) btn.BackgroundTransparency = 1 btn.Text = "" btn.ZIndex = 7 btn.Parent = f
    local function refresh() tween(track, {BackgroundColor3 = state and C.ToggleOn or C.ToggleOff}, 0.15):Play() tween(knob, {Position = state and UDim2.new(0, 18, 0.5, -6) or UDim2.new(0, 2, 0.5, -6)}, 0.15):Play() end
    btn.MouseButton1Click:Connect(function() state = not state refresh() notify(label .. ": " .. (state and "ON" or "OFF"), state and C.Success or C.Danger, 2) if callback then callback(state) end end)
    btn.MouseEnter:Connect(function() tween(f, {BackgroundTransparency = 0.1}, 0.1):Play() end) btn.MouseLeave:Connect(function() tween(f, {BackgroundTransparency = 0.3}, 0.1):Play() end)
    return f, function() return state end, function(v) state = v refresh() end
end
